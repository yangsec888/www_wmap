    # -*- encoding: utf-8 -*- VERSION = File.dirname(__FILE__) + “/version.txt”

    # Simple parser for the project version file info=Hash.new f=File.open(VERSION,'r') f.each do |line|

    line.chomp!
case line
when /^(\s)*#/
        next
when /\=/
        entry=line.split("=").map! {|x| x.strip}
        info[entry[0]]=entry[1]
end

    end f.close #

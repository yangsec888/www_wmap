    read_ver() click to toggle source

    Simple parser for the project version file

    # File lib/wmap.rb, line 42
def read_ver
        ver=Hash.new
        f=File.open(VERSION,'r')
        f.each do |line|
                line.chomp!
                case line
                when /^(\s)*#/
                        next
                when /\=/
                        entry=line.split("=").map! {|x| x.strip}
                        ver[entry[0]]=entry[1]
                end
        end
        f.close
        return ver
end
